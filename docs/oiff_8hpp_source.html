31void build(int p, int location, std::vector<Node>& tree) {
32 auto& current = tree[location];
33 ++current.count;
34 if (current.p_right - current.p_left == 1) {
35 return;
36 }
37
38 int jump;
39 if (p < current.p_mid) {
40 jump = current.jump_left;
41 if (jump < 0) {
42 jump = tree.size();
43 current.jump_left = jump;
44 tree.resize(jump + 1);
45
46 const auto& current = tree[location]; // rebind due to possible allocation.
47 auto& latest = tree.back();
48 latest.p_left = current.p_left;
49 latest.p_right = current.p_mid;
50 latest.p_mid = latest.p_left + (latest.p_right - latest.p_left) / 2;
51 }
52 } else {
53 jump = current.jump_right;
54 if (jump < 0) {
55 jump = tree.size();
56 current.jump_right = jump;
57 tree.resize(jump + 1);
58
59 const auto& current = tree[location]; // rebind due to possible allocation.
60 auto& latest = tree.back();
61 latest.p_left = current.p_mid;
62 latest.p_right = current.p_right;
63 latest.p_mid = latest.p_left + (latest.p_right - latest.p_left) / 2;
64 }
65 }
66
67 build(p, jump, tree);
68 return;
69}

71/* In this recursion, the return value is a pair containing:
72 *
73 * - The largest number of discoveries at the specified FDR threshold
74 * within the interval represented by the current node.
75 * - The number of hypotheses with p-values at or below the largest
76 * p-value in the interval represented by the current node.
77 *
78 * The first value is our desired result, but the second value needs
79 * to be returned to allow calculation of a possibly larger number of
80 * discoveries at the parent node of the current node.
81 */
82template<typename P>
83std::pair<int, int> find_boundary(int location, const std::vector<Node>& tree, const P* pvalues, int accumulated, P threshold) {
84 const auto& current = tree[location];
85 auto lowestp = pvalues[current.p_left];
86 int accumulated2 = accumulated + current.count;
87
88 // Quitting if it doesn't work at its most optimistic, i.e., assigning
89 // all points in the interval to the lowest p-value.
90 if (lowestp > threshold * accumulated2) {
91 return std::make_pair(0, accumulated2);
92 }
93
94 // Returning the cumulative number of discoveries if we're at a terminal
95 // node. This is potentially the boundary of significance for the given FDR
96 // threshold; anything with lower p-values will end up getting this
97 // adjusted p-value (or better) due to the cumulative minimum in the BH
98 // method. Of course, we still need to check with the parent to see if
99 // there might be more discoveries at an even larger p-value threshold.
100 if (current.p_right - current.p_left == 1) {
101 return std::make_pair(accumulated2, accumulated2);
102 }
103
104 // Searching the left side for the boundary. We need to do this first
105 // in order to get the cumulative sum before searching the right.
106 int best = 0;
107 if (current.jump_left >= 0) {
108 auto left = find_boundary(current.jump_left, tree, pvalues, accumulated, threshold);
109 best = left.first;
110 accumulated = left.second;
111 }
112
113 // Now checking the right side. If it gives us more discoveries, we switch
114 // 'best' to its number of discoveries. No need to update 'accumulated' as
115 // this should end up being equal to 'accumulated2' anyway.
116 if (current.jump_right >= 0) {
117 auto right = find_boundary(current.jump_right, tree, pvalues, accumulated, threshold);
118 if (right.first) {
119 best = right.first;
120 }
121 }
122
123 return std::make_pair(best, accumulated2);
124}

126template<class Iterator>
127std::vector<int> order(Iterator start, size_t n) {
128 std::vector<int> temp;
129 temp.reserve(n);
130 for (size_t i = 0; i < n; ++i) {
131 temp.push_back(i);
132 }
133
134 std::sort(temp.begin(), temp.end(), [&](int left, int right) -> bool {
135 return *(start + left) < *(start + right);
136 });
137
138 return temp;
139}

141template<typename P, typename C>
142std::pair<C, int> find_optimal_filter(size_t n_full, const P* pvalues, const C* covariates, const std::vector<int>& porder, const std::vector<int>& corder, P fdr_threshold) {
143 // Without subsetting, n_working and n_full are the same. However, we want
144 // to easily re-use the same code after subsetting, where 'porder' and
145 // 'corder' contains a subset of the indices, while 'pvalues' and
146 // 'covariates' still contain the full dataset to avoid a copy... Hence the
147 // need to distinguish between n_full and n_working in the subsetted case.
148 size_t n_working = porder.size();
149 if (!n_working) {
150 return std::pair<C, int>(0, 0);
151 }
152
153 // Finding all unique p-values and their relative ranks. Note that
154 // 'prank' is still full-length to enable easy indexing from 'corder'.
155 std::vector<int> prank(n_full, -1);
156 std::vector<P> uniq_p;
157 {
158 uniq_p.reserve(n_working);
159
160 int rank = 0;
161 auto first = *(pvalues + porder[0]);
162 if (first <= fdr_threshold) {
163 uniq_p.push_back(first);
164 prank[porder[0]] = rank;
165 }
166
167 for (size_t i = 1; i < n_working; ++i) {
168 const auto& current = *(pvalues + porder[i]);
169 if (current > fdr_threshold) {
170 break;
171 }
172 if (current != uniq_p.back()) {
173 uniq_p.push_back(current);
174 ++rank;
175 }
176 prank[porder[i]] = rank;
177 }
178 }
179
180 // Iteratively build the tree and query for the FDR threshold. Each
181 // iteration is done by increasing the covariate threshold and adding
182 // more observations with that covariate value to the tree.
183 std::vector<Node> tree(1);
184 tree.reserve(uniq_p.size() * 2);
185 tree.back().p_left = 0;
186 tree.back().p_right = uniq_p.size();
187 tree.back().p_mid = uniq_p.size() / 2;
188
189 size_t i = 0;
190 int max_hits = -1;
191 C cov_threshold = 0;
192
193 while (i < n_working) {
194 auto curcov = covariates[corder[i]];
195 do {
196 auto curprank = prank[corder[i]];
197 if (curprank >= 0) {
198 build(curprank, 0, tree);
199 }
200 ++i;
201 } while (i < n_working && covariates[corder[i]] == curcov);
202
203 // Dividing by 'i' to get the Bonferroni adjusted threshold, which is
204 // then internally multiplied by the rank to get the BH threshold.
205 auto hits = find_boundary(0, tree, uniq_p.data(), 0, fdr_threshold / i);
206
207 // Using >= so that we favor a more relaxed filter, everything else being equal.
208 if (hits.first >= max_hits) {
209 max_hits = hits.first;
210 cov_threshold = curcov;
211 }
212 }
213
214 return std::pair<C, int>(cov_threshold, max_hits);
215}

217template<class Engine>
218double quick_uniform01(Engine& eng) {
219 // Don't need to worry about checks for the edge case of returning 1
220 // due to numerical imprecision after division; this won't break the
221 // use of this function in sample().
222 return static_cast<double>(eng() - Engine::min()) / (static_cast<double>(Engine::max() - Engine::min()) + 1.0);
223}

225template<class Engine>
226void sample(size_t n, size_t s, std::vector<uint8_t>& chosen, Engine& eng) {
227 std::fill(chosen.begin(), chosen.end(), static_cast<uint8_t>(0));
228 for (size_t i = 0; i < n && s; ++i) {
229 const double threshold = static_cast<double>(s)/(n - i);
230 if (threshold >= 1 || quick_uniform01(eng) <= threshold) {
231 chosen[i] = 1;
232 --s;
233 }
234 }

320 auto porder = order(pvalues, n);
321 auto corder = order(filters, n);
322
323 std::mt19937_64 rng(random_seed);
324 size_t keep = std::ceil(n * subsample_proportion);
325 std::vector<std::pair<C, int> > output(num_iterations);
326
327 auto executor = [&](int iteration, const std::vector<uint8_t>& s, std::vector<int>& p, std::vector<int>& c) -> void {
328 slice_vector(porder, s, p);
329 slice_vector(corder, s, c);
330 output[iteration] = find_optimal_filter(n, pvalues, filters, p, c, fdr_threshold);
331 };
332
333 if (num_threads == 1) {
334 std::vector<uint8_t> selected(n);
335 std::vector<int> porder2, corder2;
336 for (int it = 0; it < num_iterations; ++it) {
337 sample(n, keep, selected, rng);
338 executor(it, selected, porder2, corder2);
339 }
340
341 } else {
342 std::vector<std::vector<uint8_t> > selected(num_threads, std::vector<uint8_t>(n));
343 std::vector<std::vector<int> > porder2(num_threads), corder2(num_threads);
344 std::vector<std::thread> threads(num_threads);
345
346 for (int it = 0; it < num_iterations; ++it) {
347 size_t thread_id = it % num_threads;
348 if (it >= num_threads) {
349 threads[thread_id].join();
350 }
351 sample(n, keep, selected[thread_id], rng); // RNG'ing is done in serial for simplicity.
352
353 threads[thread_id] = std::thread(
354 [&](int iteration, size_t tid) -> void { executor(iteration, selected[tid], porder2[tid], corder2[tid]); },
355 it, thread_id
356 );
357 }
358
359 // Joining any threads that are still running.
360 for (int t = 0; t < std::min(num_iterations, num_threads); ++t) {
361 threads[t].join();
362 }
363 }
364
365 return output;
