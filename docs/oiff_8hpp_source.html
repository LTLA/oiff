31void build(int p, int location, std::vector<Node>& tree) {
32 auto& current = tree[location];
33 ++current.count;
34 if (current.p_right - current.p_left == 1) {
35 return;
36 }
37
38 int jump;
39 if (p < current.p_mid) {
40 jump = current.jump_left;
41 if (jump < 0) {
42 jump = tree.size();
43 current.jump_left = jump;
44 tree.resize(jump + 1);
45
46 const auto& current = tree[location]; // rebind due to possible allocation.
47 auto& latest = tree.back();
48 latest.p_left = current.p_left;
49 latest.p_right = current.p_mid;
50 latest.p_mid = latest.p_left + (latest.p_right - latest.p_left) / 2;
51 }
52 } else {
53 jump = current.jump_right;
54 if (jump < 0) {
55 jump = tree.size();
56 current.jump_right = jump;
57 tree.resize(jump + 1);
58
59 const auto& current = tree[location]; // rebind due to possible allocation.
60 auto& latest = tree.back();
61 latest.p_left = current.p_mid;
62 latest.p_right = current.p_right;
63 latest.p_mid = latest.p_left + (latest.p_right - latest.p_left) / 2;
64 }
65 }
66
67 build(p, jump, tree);
68 return;
69}

71/* In this recursion, the return value is a pair containing:
72 *
73 * - The largest number of discoveries at the specified FDR threshold
74 * within the interval represented by the current node.
75 * - The number of hypotheses with p-values at or below the largest
76 * p-value in the interval represented by the current node.
77 *
78 * The first value is our desired result, but the second value needs
79 * to be returned to allow calculation of a possibly larger number of
80 * discoveries at the parent node of the current node.
81 */
82template<typename P>
83std::pair<int, int> find_boundary(int location, const std::vector<Node>& tree, const P* pvalues, int accumulated, P threshold) {
84 const auto& current = tree[location];
85 auto lowestp = pvalues[current.p_left];
86 int accumulated2 = accumulated + current.count;
87
88 // Quitting if it doesn't work at its most optimistic, i.e., assigning
89 // all points in the interval to the lowest p-value.
90 if (lowestp > threshold * accumulated2) {
91 return std::make_pair(0, accumulated2);
92 }
93
94 // Returning the cumulative number of discoveries if we're at a terminal
95 // node. This is potentially the boundary of significance for the given FDR
96 // threshold; anything with lower p-values will end up getting this
97 // adjusted p-value (or better) due to the cumulative minimum in the BH
98 // method. Of course, we still need to check with the parent to see if
99 // there might be more discoveries at an even larger p-value threshold.
100 if (current.p_right - current.p_left == 1) {
101 return std::make_pair(accumulated2, accumulated2);
102 }
103
104 // Searching the left side for the boundary. We need to do this first
105 // in order to get the cumulative sum before searching the right.
106 int best = 0;
107 if (current.jump_left >= 0) {
108 auto left = find_boundary(current.jump_left, tree, pvalues, accumulated, threshold);
109 best = left.first;
110 accumulated = left.second;
111 }
112
113 // Now checking the right side. If it gives us more discoveries, we switch
114 // 'best' to its number of discoveries. No need to update 'accumulated' as
115 // this should end up being equal to 'accumulated2' anyway.
116 if (current.jump_right >= 0) {
117 auto right = find_boundary(current.jump_right, tree, pvalues, accumulated, threshold);
118 if (right.first) {
119 best = right.first;
120 }
121 }
122
123 return std::make_pair(best, accumulated2);
124}

126template<class Iterator>
127std::vector<int> order(Iterator start, size_t n, bool reverse = false) {
128 std::vector<int> temp;
129 temp.reserve(n);
130 for (size_t i = 0; i < n; ++i) {
131 temp.push_back(i);
132 }
133
134 if (reverse) {
135 std::sort(temp.begin(), temp.end(), [&](int left, int right) -> bool {
136 return *(start + left) > *(start + right);
137 });
138 } else {
139 std::sort(temp.begin(), temp.end(), [&](int left, int right) -> bool {
140 return *(start + left) < *(start + right);
141 });
142 }
143
144 return temp;
145}

147template<class Engine>
148double quick_uniform01(Engine& eng) {
149 // Don't need to worry about checks for the edge case of returning 1
150 // due to numerical imprecision after division; this won't break the
151 // use of this function in sample().
152 return static_cast<double>(eng() - Engine::min()) / (static_cast<double>(Engine::max() - Engine::min()) + 1.0);
153}

155template<class Engine>
156void sample(size_t n, size_t s, std::vector<uint8_t>& chosen, Engine& eng) {
157 std::fill(chosen.begin(), chosen.end(), static_cast<uint8_t>(0));
158 for (size_t i = 0; i < n && s; ++i) {
159 const double threshold = static_cast<double>(s)/(n - i);
160 if (threshold >= 1 || quick_uniform01(eng) <= threshold) {
161 chosen[i] = 1;
162 --s;
163 }
164 }

260 template<typename P, typename C>
261 static Thresholds<C> find_optimal_filter(
262 size_t n_full,
263 const P* pvalues,
264 const C* covariates,
265 const std::vector<int>& porder,
266 const std::vector<int>& corder,
267 P fdr_threshold,
268 std::vector<std::pair<C, int> >& candidates)
269 {
270 // Without subsetting, n_working and n_full are the same. However, we want
271 // to easily re-use the same code after subsetting, where 'porder' and
272 // 'corder' contains a subset of the indices, while 'pvalues' and
273 // 'covariates' still contain the full dataset to avoid a copy... Hence the
274 // need to distinguish between n_full and n_working in the subsetted case.
275 size_t n_working = porder.size();
276 if (!n_working) {
277 return Thresholds<C>();
278 }
279
280 // Finding all unique p-values and their relative ranks. Note that
281 // 'prank' is still full-length to enable easy indexing from 'corder'.
282 std::vector<int> prank(n_full, -1);
283 std::vector<P> uniq_p;
284 {
285 uniq_p.reserve(n_working);
286
287 int rank = 0;
288 auto first = *(pvalues + porder[0]);
289 if (first <= fdr_threshold) {
290 uniq_p.push_back(first);
291 prank[porder[0]] = rank;
292 }
293
294 for (size_t i = 1; i < n_working; ++i) {
295 const auto& current = *(pvalues + porder[i]);
296 if (current > fdr_threshold) {
297 break;
298 }
299 if (current != uniq_p.back()) {
300 uniq_p.push_back(current);
301 ++rank;
302 }
303 prank[porder[i]] = rank;
304 }
305 }
306
307 // Iteratively build the tree and query for the FDR threshold. Each
308 // iteration is done by increasing the covariate threshold and adding
309 // more observations with that covariate value to the tree.
310 std::vector<Node> tree(1);
311 tree.reserve(uniq_p.size() * 2);
312 tree.back().p_left = 0;
313 tree.back().p_right = uniq_p.size();
314 tree.back().p_mid = uniq_p.size() / 2;
315
316 Thresholds<C> output;
317 int& max_hits = output.number;
318 candidates.clear();
319 max_hits = -1;
320 size_t i = 0;
321
322 while (i < n_working) {
323 size_t start = i;
324 auto curcov = covariates[corder[i]];
325 do {
326 auto curprank = prank[corder[i]];
327 if (curprank >= 0) {
328 build(curprank, 0, tree);
329 }
330 ++i;
331 } while (i < n_working && covariates[corder[i]] == curcov);
332
333 // Dividing by 'i' to get the Bonferroni adjusted threshold, which is
334 // then internally multiplied by the rank to get the BH threshold.
335 auto hits = find_boundary(0, tree, uniq_p.data(), 0, fdr_threshold / i);
336
337 if (hits.first >= max_hits) {
338 if (hits.first > max_hits) {
339 candidates.clear();
340 max_hits = hits.first;
341 }
342 candidates.emplace_back(curcov, i - start);
343 }
344 }
345
346 if (candidates.size()) {
347 output.first = candidates.front().first;
348 output.last = candidates.back().first;
349
350 // Computing the weighted median here.
351 int total = 0;
352 for (const auto& x : candidates) {
353 total += x.second;
354 }
355 int target = total / 2;
356
357 total = 0;
358 output.middle = output.first;
359 for (const auto& x : candidates) {
360 total += x.second;
361 if (total > target) {
362 output.middle = x.first;
363 break;
364 }
365 }
366 }
367
368 return output;
369 }

443

447 },
448 it, thread_id
449 );
450 }
451
452 // Joining any threads that are still running.
453 for (int t = 0; t < std::min(num_iterations, num_threads); ++t) {

455 }
456 }
